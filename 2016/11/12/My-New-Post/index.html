一、对象的深复制

//对象的声明
var person={
    name:"shafee",
    age:20,
    tel:[13800,10086],
    say:function(){
        console.log(this.age);
    }
}
var p = {
    gender:"male"
}
function deepClone(oldObj,newObj){
    var newObj=newObj||{};//如果传入newObj具使用该对象，否则就是用空对象
    for(var i in oldObj){//遍历原有对象属性和方法
        if(oldObj.hasOwnProperty(i)){    //判断对象本身是否有该属性
            //只拷贝原有对象的自有属性
            if(typeof oldObj[i]==="object"){
                var c= (oldObj[i].constructor==Array?[]:{});
                //使用递归
                newObj[i] = deepClone(oldObj[i],c)
            }else{
                newObj[i] =oldObj[i];//浅复制
            }

        }
    }
    return newObj;//返回一个新的oldObj所有属性跟方法的对象


}
var p1 = deepClone(person,p);//拓展p对象，并拥有person的属性跟方法
var p2 = deepClone(person);//复制person对象
